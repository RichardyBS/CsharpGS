"
  CREATE OR REPLACE EDITIONABLE PROCEDURE "RM557883"."ATUALIZAR_CAPACIDADES" 
IS
BEGIN
  UPDATE AREA A
  SET capacidade_atual = (
    SELECT COUNT(M.id_moto)
    FROM MOTO M
    JOIN VAGA V ON M.id_vaga = V.id_vaga
    JOIN CORREDOR C ON V.id_corredor = C.id_corredor
    WHERE C.id_area = A.id_area
  );

  UPDATE GALPAO G
  SET capacidade_atual = (
    SELECT SUM(A.capacidade_atual)
    FROM AREA A
    WHERE A.id_galpao = G.id_galpao
  );
END;
"

"
  CREATE OR REPLACE EDITIONABLE PROCEDURE "RM557883"."SP_ATUALIZAR_PREFERENCIA" (
    p_id_preferencia IN NUMBER,
    p_cidade IN VARCHAR2,
    p_estado IN VARCHAR2,
    p_temperatura_min IN NUMBER,
    p_temperatura_max IN NUMBER,
    p_ativo IN NUMBER
)
IS
BEGIN
    UPDATE PREFERENCIAS_NOTIFICACAO
    SET CIDADE = p_cidade,
        ESTADO = p_estado,
        TEMPERATURA_MIN = p_temperatura_min,
        TEMPERATURA_MAX = p_temperatura_max,
        ATIVO = CASE WHEN p_ativo = 1 THEN 1 ELSE 0 END,
        DATA_ATUALIZACAO = SYSDATE
    WHERE ID_PREFERENCIA = p_id_preferencia;

    IF SQL%ROWCOUNT = 0 THEN
        RAISE_APPLICATION_ERROR(-20002, 'Preferência não encontrada');
    END IF;

    COMMIT;
END;
"

"
  CREATE OR REPLACE EDITIONABLE PROCEDURE "RM557883"."SP_CONSULTAR_ALERTAS" (
    p_cidade IN VARCHAR2 DEFAULT NULL,
    p_estado IN VARCHAR2 DEFAULT NULL,
    p_alertas OUT SYS_REFCURSOR
)
IS
BEGIN
    OPEN p_alertas FOR
    SELECT 
        ID_ALERTA,
        CIDADE,
        ESTADO,
        TEMPERATURA,
        TIPO_ALERTA,
        MENSAGEM,
        DATA_HORA,
        STATUS
    FROM ALERTAS_TEMPERATURA
    WHERE (p_cidade IS NULL OR CIDADE = p_cidade)

    ORDER BY DATA_HORA DESC;
